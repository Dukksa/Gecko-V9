country_event = { #faceting failsafe
	id = echo.22
	title = echo.22.t
	desc = echo.22.d
	picture = TRADEGOODS_eventPicture
	goto = diamond_city

	fire_only_once = yes
	
	major = yes

	trigger = {
		ai = no
		has_global_flag = faceting_happened
		NOT = { has_global_flag = faceting_failsafe }
		has_institution = renaissance
		prestige = 50
		151 = {
			renaissance = 100
			OR = {
				trade_goods = glass
				trade_goods = gems
			}
			development = 15
			is_state = yes
			is_city = yes
			province_without_special_goods_produced_trigger = yes
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = echo.22.a
		151 = {
			if = {
				limit = {
					NOT = { trade_goods = gems }
				}
				change_trade_goods = gems
			}
			add_permanent_province_modifier = {
				name = diamond_district
				duration = -1
			}
		}
		set_global_flag = faceting_failsafe
	}
}
