po_show_burgundian_superiority = {
	
	category = 6 
	power_projection = no 
	power_cost_base = 0 
	prestige_base = 0.5
	ae_base = 0
	warscore_cost = {
		all_provinces = 0
		no_provinces = 50.0 
		owner_keeps = 0.0 
		conquered = 0.0 
		returned_core = 0.0 
		released_nation = 0.0 
		cancelled_subject = 0.0 
		concede_colonial_region = 0.0 
	}
	warscore_cap = 50
	requires_demand_independence = yes 
	is_make_subject = no 
	requires_is_allowed = yes 
	
	is_visible = { 
		always = yes
	}
	is_allowed = { 
		always = yes
	}
	effect = { 
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		ME_add_power_projection = { amount = 15 } 
		FROM = {
			add_prestige = -30
		}
		if = {
			limit = {
				NOT = { has_country_flag = bur_defeated_france }
				FROM = { tag = FRA }
			}
			set_country_flag = bur_defeated_france
		}
		if = {
			limit = {
				NOT = { has_country_flag = bur_defeated_emperor }
				FROM = { is_emperor = yes }
			}
			set_country_flag = bur_defeated_emperor
		}
		if = {
			limit = {
				has_country_flag = bur_defeated_emperor
				has_country_flag = bur_defeated_france
			}
			country_event = { id = ME_Burgundy_Events.10 }
			clr_country_flag = bur_defeated_emperor
			clr_country_flag = bur_defeated_france
		}
	}
	ai_weight = { 
		export_to_variable = {
			variable_name = ai_value 
			value = 10000
		}
		limit = {
			always = yes
		}
	}
}
