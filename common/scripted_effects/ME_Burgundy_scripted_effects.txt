calculate_council_support = {
	set_variable = {
		which = odds_to_accept
		value = 50
	}
	set_variable = {
		which = odds_to_reject
		value = 40
	}
	if = {
		limit = {
			has_country_flag = me_bur_council_of_war_acceptance_chance_tournament_flag
			NOT = {
				had_country_flag = {
					flag = me_bur_council_of_war_acceptance_chance_tournament_flag
					days = 1825
				}
			}
		}
		multiply_variable = {
			which = odds_to_accept
			value = 1.3
		}
	}
	if = {
		limit = {
			estate_loyalty = {
				estate = estate_nobles
				loyalty = 75
			}
		}
		multiply_variable = {
			which = odds_to_accept
			value = 1.5
		}
	}
	if = {
		limit = {
			manpower_percentage = 0.65
		}
		multiply_variable = {
			which = odds_to_accept
			value = 1.5
		}
	}
	if = {
		limit = {
			NOT = { had_recent_war = 5 }
		}
		multiply_variable = {
			which = odds_to_accept
			value = 2
		}
	}
	if = {
		limit = {
			ME_toison_dor = 1
		}
		multiply_variable = {
			which = odds_to_accept
			value = 1.25
		}
	}
	if = {
		limit = {
			ruler_has_personality = charismatic_negotiator_personality
		}
		multiply_variable = {
			which = odds_to_accept
			value = 1.1
		}
	}
	
	set_variable = {
		which = odds_to_accept_divider
		which = odds_to_accept
	}
	
	multiply_variable = {
		which = odds_to_accept_divider
		value = 100
	}
	
	change_variable = {            
		which = odds_to_reject
		which = odds_to_accept
	}
	
	divide_variable = {
		which = odds_to_accept_divider
		which = odds_to_reject
	}
	
	set_variable = {
		which = chance_to_accept
		which = odds_to_accept_divider
	}
}
