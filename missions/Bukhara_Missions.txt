	slot = 1
	generic = no
	ai = yes
	potential = {
			tag = BUK
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	buk_mission_conquer_central_asia = {
			icon = mission_arabian_fort
			required_missions = {  }
			provinces_to_highlight = {
				region = central_asia_region
				NOT = {
					country_or_non_sovereign_subject_holds = ROOT
				}
			}
			trigger = {
				central_asia_region = {
					country_or_non_sovereign_subject_holds = ROOT
				}
			}
			effect = {
				add_army_tradition = 20
				add_country_modifier = {
					name = "integrating_the_steppe_peoples"
					duration = -1
				}
				mongolia_region = {
					limit = {
						NOT = { is_permanent_claim = ROOT	}
						NOT = { is_core = ROOT	}
					}
					add_permanent_claim = ROOT
				}	
			}
	}
	buk_mission_conquer_mongolia = {
		icon = mission_chinese_general_riding
		required_missions = { buk_mission_conquer_central_asia }
		provinces_to_highlight = {
			region = mongolia_region
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
			NOT = {
				is_empty = yes
			}
		}
		trigger = {
			owns_core_province = 2190
			custom_trigger_tooltip = {
				tooltip = mongol_china_mission_tooltip
				mongolia_region = {
					type = all
					OR = {
						is_empty = yes
						country_or_non_sovereign_subject_holds = ROOT
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					NOT = {
						government_rank = 3
					}
				}
				set_government_rank = 3
			}
			2190 = {
				add_base_tax = 2
				add_base_production = 2
				add_base_manpower = 2
				change_culture = ROOT
			}
			add_country_modifier = {
				name = "fall_of_china"
				duration = 7300
			}
		}
	}
}
